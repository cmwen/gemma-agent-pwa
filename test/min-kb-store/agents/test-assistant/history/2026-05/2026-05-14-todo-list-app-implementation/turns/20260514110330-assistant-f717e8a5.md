### Final Implementation (Python)

This single script incorporates all three required features: in-memory management (Step 1), file persistence (Step 2), and task completion/deletion (Step 3).

```python
import json
import os

FILE_NAME = "todo_list.json"

def load_tasks():
    """Loads tasks from the JSON file."""
    if os.path.exists(FILE_NAME):
        try:
            with open(FILE_NAME, 'r') as f:
                return json.load(f)
        except json.JSONDecodeError:
            # Handle case where file exists but is empty or corrupted
            return []
    return []

def save_tasks(tasks):
    """Saves the list of tasks to the JSON file."""
    with open(FILE_NAME, 'w') as f:
        json.dump(tasks, f, indent=4)

# --- Step 1: Core Data Structure and Basic CLI Functions ---

def add_task(tasks):
    """Adds a new task to the list."""
    task_description = input("Enter the new task description: ")
    if task_description.strip():
        tasks.append({"description": task_description, "completed": False})
        print(f"Task added: '{task_description}'")
    else:
        print("Task description cannot be empty.")

def view_tasks(tasks):
    """Displays all tasks in the list."""
    if not tasks:
        print("\nYour to-do list is empty!")
        return

    print("\n--- Current To-Do List ---")
    for index, task in enumerate(tasks):
        status = "[DONE]" if task["completed"] else "[TODO]"
        print(f"{index + 1}. {status} {task['description']}")
    print("----------------")

# --- Step 2: Data Persistence (Save and Load) ---

def load_data():
    """Loads tasks from the file."""
    return load_tasks()

def save_data(tasks):
    """Saves tasks to the file."""
    save_tasks(tasks)

# --- Step 3: Completion and Deletion Functionality ---

def complete_task(tasks, task_index):
    """Marks a task as complete."""
    if 0 <= task_index < len(tasks):
        tasks[task_index]["completed"] = True
        print(f"Task {task_index + 1} marked as complete.")
    else:
        print("Error: Invalid task number.")

def delete_task(tasks, task_index):
    """Deletes a task from the list."""
    if 0 <= task_index < len(tasks):
        removed_task = tasks.pop(task_index)
        print(f"Task deleted: '{removed_task['description']}'")
    else:
        print("Error: Invalid task number.")

# --- Main Application Loop ---

def main():
    """Main function to run the To-Do List application."""
    tasks = load_data()
    print("To-Do List Application Initialized.")

    while True:
        print("\n===== Menu ===")
        print("1. Add Task")
        print("2. View Tasks")
        print("3. Complete Task")
        print("4. Delete Task")
        print("5. Save & Exit")
        
        choice = input("Select an option (1-5): ")

        if choice == '1':
            add_task(tasks)
        elif choice == '2':
            view_tasks(tasks)
        elif choice == '3':
            if not tasks:
                print("List is empty. Nothing to complete.")
                continue
            view_tasks(tasks)
            try:
                task_num = int(input("Enter the number of the task to mark as complete: ")) - 1
                complete_task(tasks, task_num)
            except ValueError:
                print("Invalid input. Please enter a number.")
        elif choice == '4':
            if not tasks:
                print("List is empty. Nothing to delete.")
                continue
            view_tasks(tasks)
            try:
                task_num = int(input("Enter the number of the task to delete: ")) - 1
                delete_task(tasks, task_num)
            except ValueError:
                print("Invalid input. Please enter a number.")
        elif choice == '5':
            save_data(tasks)
            print("Data saved. Exiting application. Goodbye!")
            break
        else:
            print("Invalid selection. Please choose an option from 1 to 5.")

if __name__ == "__main__":
    main()
```
